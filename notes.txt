# dict 字典，键值对，无序，keyo必须不可变
d = {
    'Adam':95,
    'Lisa':85
}
len(d) # length of d
if 'Adam' in d: # 判断是否在集合中
    d['Adam'] # 下标访问
# 另外一种访问方法
d.get('Adam') # 访问失败返回None
#dict和list相比占内存，查询速度快，查询时间不随尺寸增长而增长
d['NewGuy']=70 # 直接添加新元素，已有的话就替换
for a in d:
    print(a,d[a])
